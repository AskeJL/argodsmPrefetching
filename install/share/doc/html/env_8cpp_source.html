198 template<typename T>
199 std::pair<bool, T> parse_env(std::string name, T fallback) {
200  auto env_get = std::getenv(name.c_str());
201  try {
202  if(env_get != nullptr) {
203  std::string env_string(env_get);
204  std::stringstream env_stream(env_string);
205  T env_value;
206  env_stream >> env_value;
207  return std::make_pair(true, env_value);
208  } else {
209  return std::make_pair(false, fallback);
210  }
211  } catch (const std::invalid_argument& e) {
212  // environment variable exists, but value is not convertable to an unsigned long
213  std::cerr << msg_illegal_format << name << std::endl;
214  throw;
215  } catch (const std::out_of_range& e) {
216  // environment variable exists, but value is out of range
217  std::cerr << msg_out_of_range << name << std::endl;
218  throw;
219  }
