130  void write_back_index(std::size_t cache_index) {
131  cache_locks[cache_index].lock();
132  assert(cacheControl[cache_index].dirty == DIRTY);
133  const std::uintptr_t page_address = cacheControl[cache_index].tag;

141  storepageDIFF(cache_index+i, page_size*i+page_address);
