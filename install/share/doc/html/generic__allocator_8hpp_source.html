222  // by max_align_t to ensure that all allocations are well aligned
223  std::size_t aligned_alloc_size = align_up(n * sizeof(T), alignof(std::max_align_t));

227  std::size_t aligned_n = aligned_alloc_size / sizeof(T);

231  T* allocation = freelist[aligned_n].top();
232  freelist[aligned_n].pop();
