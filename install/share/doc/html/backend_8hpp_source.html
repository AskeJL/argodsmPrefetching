566 T fetch_add(global_ptr<T> obj, U value, memory_order order = memory_order::acq_rel) {
567  T out_buffer;
568  // We need a buffer with the same size as T
569  static_assert(std::is_convertible<U, T>::value,
570  "It is not possible to implicitly convert \'desired\' to an"
571  " object of type T.");
572  T value_buffer(value);
573  static_assert(std::is_trivially_copy_assignable<T>::value,
574  "T must be trivially copy assignable");
575 
576  if (order == memory_order::acq_rel || order == memory_order::release)
577  release();
578 
579  // The order is important here, as floats are signed as well
580  if (std::is_floating_point<T>::value)
581  _fetch_add_float(global_ptr<void>(obj), &value_buffer, sizeof(T), &out_buffer);
582  else if (std::is_unsigned<T>::value)
583  _fetch_add_uint(global_ptr<void>(obj), &value_buffer, sizeof(T), &out_buffer);
584  else if (std::is_signed<T>::value)
585  _fetch_add_int(global_ptr<void>(obj), &value_buffer, sizeof(T), &out_buffer);
586  else
587  throw std::invalid_argument("Invalid type V for value");
588 
589  if (order == memory_order::acq_rel || order == memory_order::acquire)
590  acquire();
591 
592  return out_buffer;
593 }
