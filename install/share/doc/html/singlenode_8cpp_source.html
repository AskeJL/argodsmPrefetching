153 void barrier(std::size_t threadcount, upgrade_type upgrade) {
154  (void)upgrade;
155  /* initially: flag = false */
156  std::unique_lock<std::mutex> barrier_lock(barrier_mutex);
157  barrier_counter++;
158  if(barrier_counter == threadcount) {
159  /* avoid progress if some thread is still stuck in the previous barrier */
160  barrier_cv.wait(barrier_lock, []{ return !barrier_flag.load(); });
161  /* all threads in this barrier now,
162  * flip flag to avoid progress in next barrier */
163  barrier_flag = true;
164  barrier_cv.notify_all();
165  }
166 
167  /* wait for flag to flip to ensure all threads have reached this barrier */
168  barrier_cv.wait(barrier_lock, []{ return barrier_flag.load(); });
169 
170  /* all threads synchronized here. cleanup needed. */
171 
172  barrier_counter--;
173  if(barrier_counter == 0) {
174  /* all threads successfully left the wait call, allow next barrier */
175  barrier_flag = false;
176  barrier_cv.notify_all(); // other threads + unstick next barrier's final thread
177  }
178 
183  barrier_cv.wait(barrier_lock, []{ return !barrier_flag; });
184 }
