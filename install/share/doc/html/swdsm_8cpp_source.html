625  sharers = globalSharers[classidx];
626  });

629  if(writers != id && isZeroOrPowerOf2(writers)) {

631  globalSharers[classidx+1] |= id; // register locally

663  [&](std::size_t win_index) {

665  1, MPI_LONG, MPI_BOR, sharer_windows[win_index][owner]);
666  });
667  } else {

669  for(argo::num_nodes_t n = 0; n < numtasks; n++) {
670  if(n != workrank && ((static_cast<std::uint64_t>(1) << n)&sharers) != 0) {
671  sharer_op(MPI_LOCK_EXCLUSIVE, n, classidx+1,
672  [&](std::size_t win_index) {
673  MPI_Accumulate(&id, 1, MPI_LONG, n, sharer_win_offset+1,
674  1, MPI_LONG, MPI_BOR, sharer_windows[win_index][n]);
675  });
676  }
677  }
678  }
679  }

684  cache_locks[startIndex].unlock();
685  double t2 = MPI_Wtime();
686  argo_write_buffer->add(startIndex);
687  std::lock_guard<std::mutex> store_lock(stats.store_time_mutex);
688  stats.store_time += t2-t1;
689  return;

951 
952  double t1 = MPI_Wtime();
953  for(std::size_t i = 0; i < cachesize; i += CACHELINE) {

957  lineAddr*=(pagesize*CACHELINE);
958  std::size_t classidx = get_classification_index(lineAddr);
959  argo_byte dirty = cacheControl[i].dirty;
960 
961  if(flushed == 0 && dirty == DIRTY) {
962  argo_write_buffer->flush();
963  flushed = 1;

965  std::size_t win_index = get_sharer_win_index(classidx);
966  mpi_lock_sharer[win_index][workrank].lock(MPI_LOCK_SHARED, workrank, sharer_windows[win_index][workrank]);
967  if(
968  // node is single writer
969  (globalSharers[classidx+1] == id)
970  ||
971  // No writer and assert that the node is a sharer
972  ((globalSharers[classidx+1] == 0) && ((globalSharers[classidx]&id) == id))
973  ) {
974  mpi_lock_sharer[win_index][workrank].unlock(workrank, sharer_windows[win_index][workrank]);
975  touchedcache[i] = 1;
976  /* nothing - we keep the pages, SD is done in flushWB */
977  } else {

979  mpi_lock_sharer[win_index][workrank].unlock(workrank, sharer_windows[win_index][workrank]);
980  cacheControl[i].dirty = CLEAN;
981  cacheControl[i].state = INVALID;
982  touchedcache[i] = 0;
983  mprotect(static_cast<char*>(startAddr) + lineAddr, pagesize*CACHELINE, PROT_NONE);
984  }
985  }
986  }

1037 void swdsm_argo_barrier(int n, argo::backend::upgrade_type upgrade) {
1038  pthread_t barrierlockholder;
1039  double t1 = MPI_Wtime();
1040 
1041  // Wait for n threads to arrive
1042  pthread_barrier_wait(&threadbarrier[n]);
1043 
1044  // Optimize the single node case
1045  if(argo_get_nodes() == 1) {
1046  stats.barriers++;
1047  stats.barriertime += MPI_Wtime() - t1;
1048  return;
1049  }
1050 
1051  // Let one thread per node perform MPI operations
1052  if(pthread_mutex_trylock(&barriermutex) == 0) {
1053  barrierlockholder = pthread_self();
1054  std::unique_lock lock(sync_lock);
1055 
1056  // Perform SD followed by SI
1057  argo_write_buffer->flush();
1058  MPI_Barrier(argo_comm);
1059  self_invalidation();
1060 
1061  // Perform upgrade if requested
1062  if (upgrade != argo::backend::upgrade_type::upgrade_none) {
1063  self_upgrade(upgrade);
1064  MPI_Barrier(argo_comm);
1065  }
1066  }
1067 
1068  // Wait for n threads to arrive
1069  pthread_barrier_wait(&threadbarrier[n]);
1070  if (pthread_equal(barrierlockholder, pthread_self())) {
1071  pthread_mutex_unlock(&barriermutex);
1072  stats.barriers++;
1073  stats.barriertime += MPI_Wtime() - t1;
1074  }
